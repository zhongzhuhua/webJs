定义方法可以使用下列方法
【函数声明】
function name() {}
【函数表达式】
// 匿名表达式
var name = function() {};
// 命名表达式，bug比较多，不建议用，例如在 ie6-8 name !== myName，其他高级浏览器里面，会报错
var name = function myName() {};
// 立即执行表达式
(function(){})();
// 返回值表达式
return function() {};
【函数构造器】new Function(形参,'执行语句'); 比较少用
var fun3 = new Function('a', 'b', 'console.log(a + b)');
fun3(1, 2); 会打印出 3

三者有区别，使用函数声明来定义的函数，会被顶置到其他语句之前
